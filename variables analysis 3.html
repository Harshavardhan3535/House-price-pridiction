# Function to analyze a single variable
def analyze_variable(df, variable):
    plt.figure(figsize=(18, 6))

    # Distribution Plot
    plt.subplot(1, 3, 1)
    sns.histplot(df[variable], kde=True, bins=30)
    plt.title(f'Distribution of {variable}')

    # Box Plot for Outliers
    plt.subplot(1, 3, 2)
    sns.boxplot(x=df[variable])
    plt.title(f'Box Plot of {variable}')

    # Scatter Plot (if applicable)
    if variable != 'SalePrice':  # Skip if the variable is 'SalePrice'
        plt.subplot(1, 3, 3)
        sns.scatterplot(x=df[variable], y=df['SalePrice'])
        plt.title(f'{variable} vs SalePrice')
        plt.xlabel(variable)
        plt.ylabel('SalePrice')

    plt.tight_layout()
    plt.show()

# List of variables to analyze
variables_to_analyze = ['GrLivArea', 'TotalBsmtSF', '1stFlrSF', 'LotArea', 'OverallQual']

# Analyze each variable
for var in variables_to_analyze:
    analyze_variable(train_df, var)
